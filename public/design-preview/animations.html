/* Curved math path coordinates positioning cache + tangent rotate dynamic alignment on scroll */
// SVG Path D-Attribute: M 230 0 C 420 300, 40 500, 230 800 C 420 1100, 40 1300, 230 1600 C 420 1900, 40 2100, 230 2400 C 420 2700, 40 2900, 230 3200

// On page load: cache points and tangent angles directly from SVG path
railwayCards.forEach((card) => {
  const dist = parseFloat(card.dataset.offset) * pathLength;
  const pt = path.getPointAtLength(dist);
  const ptNext = path.getPointAtLength(Math.min(dist + 6, pathLength));
  card.baseX = pt.x;
  card.baseY = pt.y;
  card.baseAngle = Math.atan2(ptNext.y - pt.y, ptNext.x - pt.x) * (180 / Math.PI) - 90;
});

// During Scroll event loop:
scroller.addEventListener("scroll", () => {
  const viewportCenter = scroller.scrollTop + scroller.clientHeight / 2;
  
  railwayCards.forEach((card) => {
    const distanceToCenter = card.baseY - viewportCenter;
    const focusFactor = Math.max(0, 1 - Math.abs(distanceToCenter) / 220);
    const easeFocus = Math.sin(focusFactor * Math.PI / 2);
    
    // Scale up, drift perpendicularly, and rotate kinetic tangent
    const scale = 0.92 + easeFocus * 0.12;
    const opacity = 0.25 + easeFocus * 0.75;
    const driftX = (distanceToCenter / 220) * 20; 
    const sway = (distanceToCenter / 220) * 12;
    const finalRotate = card.baseAngle + sway * (1 - easeFocus);
    
    card.style.opacity = opacity;
    card.style.left = `${card.baseX}px`;
    card.style.top = `${card.baseY}px`;
    card.style.transform = `translate(-50%, -50%) translate3d(${driftX}px, 0, 0) scale(${scale}) rotate(${finalRotate}deg)`;
  });
});
